1 Create the Base API Client

Create api_client.py. Write the class signature and let Copilot build the infrastructure:

api_client.py
# Base API client with retry logic, rate limiting, and error handling

import requests
import time
import logging
from typing import Optional, Dict, Any
from dataclasses import dataclass, field
from urllib.parse import urljoin

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class APIError(Exception):
    """Custom exception for API errors."""
    def __init__(self, status_code: int, message: str):
        self.status_code = status_code
        self.message = message
        super().__init__(f"API Error {status_code}: {message}")


class BaseAPIClient:
    """Reusable base class for REST API clients with built-in resilience."""

    def __init__(self, base_url: str, api_key: Optional[str] = None,
                 max_retries: int = 3, timeout: int = 30):
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key
        self.max_retries = max_retries
        self.timeout = timeout
        self.session = requests.Session()
        self._setup_headers()

    def _setup_headers(self):
        """Set default headers. Override in subclasses."""
        self.session.headers.update({
            "Accept": "application/json",
            "User-Agent": "CopilotLabAPIClient/1.0"
        })

    def _request(self, method: str, endpoint: str,
                 params: Dict = None, data: Dict = None) -> Dict:
        """Make an HTTP request with retry logic and error handling."""
        url = f"{self.base_url}/{endpoint.lstrip('/')}"

        for attempt in range(self.max_retries):
            try:
                response = self.session.request(
                    method=method,
                    url=url,
                    params=params,
                    json=data,
                    timeout=self.timeout
                )

                # Handle rate limiting (429)
                if response.status_code == 429:
                    retry_after = int(response.headers.get("Retry-After", 60))
                    logger.warning(f"Rate limited. Waiting {retry_after}s...")
                    time.sleep(retry_after)
                    continue

                # Handle server errors (5xx) with exponential backoff
                if response.status_code >= 500:
                    wait = 2 ** attempt
                    logger.warning(f"Server error {response.status_code}. "
                                   f"Retry {attempt+1}/{self.max_retries} in {wait}s")
                    time.sleep(wait)
                    continue

                # Raise for client errors (4xx)
                if response.status_code >= 400:
                    raise APIError(response.status_code, response.text)

                return response.json()

            except requests.ConnectionError as e:
                if attempt == self.max_retries - 1:
                    raise
                logger.warning(f"Connection error, retrying... ({attempt+1})")
                time.sleep(2 ** attempt)

        raise APIError(0, "Max retries exceeded")

    def get(self, endpoint: str, params: Dict = None) -> Dict:
        return self._request("GET", endpoint, params=params)

    def post(self, endpoint: str, data: Dict = None) -> Dict:
        return self._request("POST", endpoint, data=data)

# GitHub API client — fetch repos, users, and issues

from api_client import BaseAPIClient
from dataclasses import dataclass
from typing import List, Optional
from datetime import datetime


@dataclass
class Repository:
    """Data model for a GitHub repository."""
    name: str
    full_name: str
    description: Optional[str]
    stars: int
    forks: int
    language: Optional[str]
    url: str
    created_at: str
    updated_at: str


@dataclass
class GitHubUser:
    """Data model for a GitHub user."""
    login: str
    name: Optional[str]
    bio: Optional[str]
    public_repos: int
    followers: int
    following: int
    avatar_url: str


class GitHubClient(BaseAPIClient):
    """Client for the GitHub REST API."""

    def __init__(self, token: Optional[str] = None):
        super().__init__("https://api.github.com", api_key=token)
        if token:
            self.session.headers["Authorization"] = f"token {token}"

    def get_user(self, username: str) -> GitHubUser:
        """Fetch a GitHub user's profile."""
        data = self.get(f"/users/{username}")
        return GitHubUser(
            login=data["login"],
            name=data.get("name"),
            bio=data.get("bio"),
            public_repos=data["public_repos"],
            followers=data["followers"],
            following=data["following"],
            avatar_url=data["avatar_url"]
        )

    def get_repos(self, username: str, sort: str = "updated",
                  per_page: int = 10) -> List[Repository]:
        """Fetch a user's repositories."""
        data = self.get(f"/users/{username}/repos", params={
            "sort": sort,
            "per_page": per_page,
            "direction": "desc"
        })

                url=r["html_url"],

                url=r["html_url"],
                created_at=r["created_at"],
                updated_at=r["updated_at"]
            )
            for r in data["items"]
        ]
3 Build a Weather API Client

Create weather_client.py. This demonstrates how Copilot handles API key authentication and data transformation:

weather_client.py
# Weather API client using OpenWeatherMap API pattern
# Get your free API key at: https://openweathermap.org/api

from api_client import BaseAPIClient
from dataclasses import dataclass
from typing import List, Optional


@dataclass
class Weather:
    """Current weather data."""
    city: str
    country: str
    temp_celsius: float
    feels_like: float
    humidity: int
    description: str
    wind_speed: float
    icon: str

    @property
    def temp_fahrenheit(self) -> float:
        return round(self.temp_celsius * 9/5 + 32, 1)

    def __str__(self):
        return (f"🌍 {self.city}, {self.country}\n"
                f"🌡️  {self.temp_celsius}°C ({self.temp_fahrenheit}°F)\n"
                f"💧 Humidity: {self.humidity}%\n"
                f"🌬️  Wind: {self.wind_speed} m/s\n"
                f"☁️  {self.description}")


class WeatherClient(BaseAPIClient):
    """Client for the OpenWeatherMap API."""

    def __init__(self, api_key: str):
        super().__init__("https://api.openweathermap.org/data/2.5",
                         api_key=api_key)

    def get_current(self, city: str) -> Weather:
        """Get current weather for a city."""
        data = self.get("/weather", params={
            "q": city,
            "appid": self.api_key,
            "units": "metric"
        })
        return Weather(
            city=data["name"],
            country=data["sys"]["country"],
            temp_celsius=data["main"]["temp"],
            feels_like=data["main"]["feels_like"],
            humidity=data["main"]["humidity"],
            description=data["weather"][0]["description"].title(),
            wind_speed=data["wind"]["speed"],
            icon=data["weather"][0]["icon"]
        )

    def get_forecast(self, city: str, days: int = 5) -> List[Weather]:
        """Get weather forecast for upcoming days."""
        data = self.get("/forecast", params={
            "q": city,
            "appid": self.api_key,
            "units": "metric",
            "cnt": days * 8  # API returns 3-hour intervals
        })
        # Get one reading per day (noon)
        daily = [item for item in data["list"]
                 if "12:00:00" in item["dt_txt"]]
        return [
            Weather(
                city=data["city"]["name"],
                country=data["city"]["country"],
                temp_celsius=d["main"]["temp"],
                feels_like=d["main"]["feels_like"],
                humidity=d["main"]["humidity"],
                description=d["weather"][0]["description"].title(),
                wind_speed=d["wind"]["speed"],
                icon=d["weather"][0]["icon"]
            )
            for d in daily
        ]
4 Test the Clients

Create main.py to test everything. The GitHub API works without a key (lower rate limit):

main.py
from github_client import GitHubClient

def demo_github():
    """Demo the GitHub API client."""
    client = GitHubClient()  # No token = public access

    # Fetch a user profile
    user = client.get_user("torvalds")
    print(f"👤 {user.name} (@{user.login})")
    print(f"   📦 {user.public_repos} repos | "
          f"👥 {user.followers} followers")

    # Fetch their repos
    repos = client.get_repos("torvalds", sort="stars", per_page=5)
    print(f"\n📦 Top {len(repos)} repos:")
    for repo in repos:
        print(f"   ⭐ {repo.stars:,} | {repo.name} — {repo.description}")

    # Search for repos
    print("\n🔍 Searching for 'machine learning' in Python...")
    results = client.search_repos("machine learning",
                                   language="python", min_stars=1000)
    for r in results[:5]:
        print(f"   ⭐ {r.stars:,} | {r.full_name}")

if __name__ == "__main__":
    demo_github()

  • /explain — Copilot will break down the retry logic, exponential backoff, and error handling
