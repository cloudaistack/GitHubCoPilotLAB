# Weather Dashboard — Copilot Instructions

## Code Style
- Use ES6+ features (const/let, arrow functions, template literals, destructuring)
- Use JSDoc comments for all public functions
- Prefer async/await over .then() chains
- Use descriptive variable names (not single letters)
- Handle errors with try/catch blocks

## Architecture
- Each JS module should export its public API as an object
- weatherApi.js must include a mock data fallback when no API key is set
- ui.js should never call the API directly — it receives data as arguments
- storage.js should validate data before saving to localStorage

## Design
- Use CSS custom properties for all colors
- All measurements should use rem units
- Cards should use backdrop-filter for glassmorphism effect
- Animations should respect prefers-reduced-motion
💡 Why This Matters

With these instructions in place, every Copilot suggestion in this project will follow your team's standards. Try writing a function without these instructions, then with — you'll see the difference in suggestion quality.

3 Phase 3: Build the Weather API Module

Open js/weatherApi.js. Write these comments and let Copilot implement each function:

js/weatherApi.js
/**
 * Weather API module with mock data fallback.
 * Uses OpenWeatherMap API when API_KEY is set,
 * otherwise returns realistic mock data for development.
 */

const API_KEY = ""; // Leave empty to use mock data
const BASE_URL = "https://api.openweathermap.org/data/2.5";

// Mock data for development without an API key
const MOCK_WEATHER = {
  "London":    { temp: 14, feels: 12, humidity: 72, wind: 5.2, desc: "Overcast Clouds", icon: "04d" },
  "New York":  { temp: 28, feels: 31, humidity: 65, wind: 3.8, desc: "Partly Cloudy", icon: "02d" },
  "Tokyo":     { temp: 26, feels: 29, humidity: 80, wind: 2.1, desc: "Light Rain", icon: "10d" },
  "Sydney":    { temp: 18, feels: 16, humidity: 55, wind: 6.7, desc: "Clear Sky", icon: "01d" },
  "Mumbai":    { temp: 32, feels: 38, humidity: 85, wind: 4.5, desc: "Thunderstorm", icon: "11d" },
  "Paris":     { temp: 20, feels: 19, humidity: 60, wind: 3.2, desc: "Scattered Clouds", icon: "03d" }
};

/**
 * Get current weather for a city.
 * @param {string} city - City name
 * @returns {Promise<Object>} Weather data
 */
async function getCurrentWeather(city) {
  if (!API_KEY) {
    return getMockWeather(city);
  }

  try {
    const response = await fetch(
      `${BASE_URL}/weather?q=${encodeURIComponent(city)}&appid=${API_KEY}&units=metric`
    );

    if (!response.ok) {
      throw new Error(`City not found: ${city}`);
    }

    const data = await response.json();
    return {
      city: data.name,
      country: data.sys.country,
      temp: Math.round(data.main.temp),
      feelsLike: Math.round(data.main.feels_like),
      humidity: data.main.humidity,
      windSpeed: data.wind.speed,
      description: data.weather[0].description,
      icon: data.weather[0].icon,
      timestamp: new Date()
    };
  } catch (error) {
    console.error(`Error fetching weather for ${city}:`, error);
    return getMockWeather(city); // Graceful fallback
  }
}

/**
 * Get 5-day forecast for a city.
 * @param {string} city - City name
 * @returns {Promise<Array>} Array of daily forecast objects
 */
async function getForecast(city) {
  if (!API_KEY) {
    return getMockForecast(city);
  }

  const response = await fetch(
    `${BASE_URL}/forecast?q=${encodeURIComponent(city)}&appid=${API_KEY}&units=metric`
  );
  const data = await response.json();

  // Group by day and take noon reading
  const daily = data.list.filter(item => item.dt_txt.includes("12:00:00"));
  return daily.map(d => ({
    date: new Date(d.dt * 1000),
    temp: Math.round(d.main.temp),
    tempMin: Math.round(d.main.temp_min),
    tempMax: Math.round(d.main.temp_max),
    description: d.weather[0].description,
    icon: d.weather[0].icon,
    humidity: d.main.humidity
  }));
}

function getMockWeather(city) {
  const mock = MOCK_WEATHER[city] || MOCK_WEATHER["London"];
  return {
    city: city,
    country: "--",
    temp: mock.temp,
    feelsLike: mock.feels,
    humidity: mock.humidity,
    windSpeed: mock.wind,
    description: mock.desc,
    icon: mock.icon,
    timestamp: new Date(),
    isMock: true
  };
}

function getMockForecast(city) {
  const base = MOCK_WEATHER[city] || MOCK_WEATHER["London"];
  return Array.from({ length: 5 }, (_, i) => ({
    date: new Date(Date.now() + (i + 1) * 86400000),
    temp: base.temp + Math.floor(Math.random() * 6) - 3,
    tempMin: base.temp - 3,
    tempMax: base.temp + 3,
    description: base.desc,
    icon: base.icon,
    humidity: base.humidity + Math.floor(Math.random() * 10) - 5
  }));
}

// Export public API
const WeatherAPI = { getCurrentWeather, getForecast };
4 Phase 4: Build the Storage Module

Open js/storage.js and write the comment header. Let Copilot complete each function:

js/storage.js
/**
 * LocalStorage manager for favorite cities and cached weather data.
 */

const FAVORITES_KEY = "weather_favorites";
const CACHE_KEY = "weather_cache";
const CACHE_TTL = 600000; // 10 minutes in ms

const Storage = {
  /** Get list of favorite cities */
  getFavorites() {
    try {
      const data = localStorage.getItem(FAVORITES_KEY);
      return data ? JSON.parse(data) : [];
    } catch {
      return [];
    }
  },

  /** Add a city to favorites (max 10) */
  addFavorite(city) {
    const favorites = this.getFavorites();
    const normalized = city.trim();
    if (!normalized || favorites.includes(normalized)) return false;
    if (favorites.length >= 10) favorites.shift(); // Remove oldest
    favorites.push(normalized);
    localStorage.setItem(FAVORITES_KEY, JSON.stringify(favorites));
    return true;
  },

  /** Remove a city from favorites */
  removeFavorite(city) {
    const favorites = this.getFavorites().filter(c => c !== city);
    localStorage.setItem(FAVORITES_KEY, JSON.stringify(favorites));
  },

  /** Check if city is a favorite */
  isFavorite(city) {
    return this.getFavorites().includes(city);
  },

  /** Cache weather data with TTL */
  cacheWeather(city, data) {
    try {
      const cache = JSON.parse(localStorage.getItem(CACHE_KEY) || "{}");
      cache[city] = { data, timestamp: Date.now() };
      localStorage.setItem(CACHE_KEY, JSON.stringify(cache));
    } catch (e) {
      console.warn("Cache write failed:", e);
    }
  },

  /** Get cached weather if not expired */
  getCachedWeather(city) {
    try {
      const cache = JSON.parse(localStorage.getItem(CACHE_KEY) || "{}");
      const entry = cache[city];
      if (entry && (Date.now() - entry.timestamp) < CACHE_TTL) {
        return entry.data;
      }
    } catch { }
    return null;
  }
};

/**
 * UI rendering module — handles all DOM manipulation.
 * Never calls the API directly; receives data as arguments.
 */

const WEATHER_ICONS = {
  "01d": "☀️", "01n": "🌙",
  "02d": "⛅", "02n": "☁️",
  "03d": "☁️", "03n": "☁️",
  "04d": "☁️", "04n": "☁️",
  "09d": "🌧️", "09n": "🌧️",
  "10d": "🌦️", "10n": "🌧️",
  "11d": "⛈️", "11n": "⛈️",
  "13d": "❄️", "13n": "❄️",
  "50d": "🌫️", "50n": "🌫️"
};

const UI = {
  /** Render the main weather card */
  renderCurrentWeather(weather) {
    const container = document.getElementById("current-weather");
    const icon = WEATHER_ICONS[weather.icon] || "🌍";

    container.innerHTML = `
      <div class="weather-card weather-card--current">
        <div class="weather-card__icon">${icon}</div>
        <h2 class="weather-card__city">${weather.city}</h2>
        <p class="weather-card__country">${weather.country}</p>
        <div class="weather-card__temp">${weather.temp}°C</div>
        <p class="weather-card__desc">${weather.description}</p>
        <div class="weather-card__details">
          <div>💧 ${weather.humidity}%</div>
          <div>🌬️ ${weather.windSpeed} m/s</div>
          <div>🌡️ Feels ${weather.feelsLike}°C</div>
        </div>
        ${weather.isMock ? '<p class="weather-card__mock">📋 Mock Data</p>' : ''}
      </div>
    `;

    container.querySelector(".weather-card").classList.add("fade-in");
  },

  /** Render the 5-day forecast */
  renderForecast(forecast) {
    const container = document.getElementById("forecast");
    const days = ["Sun","Mon","Tue","Wed","Thu","Fri","Sat"];

    container.innerHTML = forecast.map((day, i) => {
      const date = new Date(day.date);
      const icon = WEATHER_ICONS[day.icon] || "🌍";
      return `
        <div class="forecast-card fade-in" style="animation-delay:${i * 0.1}s">
          <div class="forecast-card__day">${days[date.getDay()]}</div>
          <div class="forecast-card__icon">${icon}</div>
          <div class="forecast-card__temp">${day.temp}°C</div>
          <div class="forecast-card__range">${day.tempMin}° / ${day.tempMax}°</div>
          <div class="forecast-card__desc">${day.description}</div>
        </div>
      `;
    }).join("");
  },

  /** Render favorites list */
  renderFavorites(favorites, onCityClick) {
    const container = document.getElementById("favorites");
    if (!favorites.length) {
      container.innerHTML = '<p class="no-favorites">No favorites yet. Click ⭐ to add.</p>';
      return;
    }
    container.innerHTML = favorites.map(city => `
      <button class="favorite-btn" data-city="${city}">${city}</button>
    `).join("");

    container.querySelectorAll(".favorite-btn").forEach(btn => {
      btn.addEventListener("click", () => onCityClick(btn.dataset.city));
    });
  },

  /** Show loading spinner */
  showLoading() {
    document.getElementById("current-weather").innerHTML =
      '<div class="loading-spinner"></div>';
  },

  /** Show error message */
  showError(message) {
    document.getElementById("current-weather").innerHTML = `
      <div class="error-card">
        <span class="error-card__icon">⚠️</span>
        <p>${message}</p>
      </div>
    `;
  }
};

js/app.js
/**
 * Main application controller.
 * Wires together WeatherAPI, Storage, and UI modules.
 */

document.addEventListener("DOMContentLoaded", () => {
  const searchForm = document.getElementById("search-form");
  const searchInput = document.getElementById("search-input");
  const favBtn = document.getElementById("fav-btn");
  let currentCity = "London";

  /** Fetch and display weather for a city */
  async function loadWeather(city) {
    UI.showLoading();
    currentCity = city;

    // Check cache first
    const cached = Storage.getCachedWeather(city);
    if (cached) {
      UI.renderCurrentWeather(cached.current);
      UI.renderForecast(cached.forecast);
      updateFavButton();
      return;
    }

    try {
      const [current, forecast] = await Promise.all([
        WeatherAPI.getCurrentWeather(city),
        WeatherAPI.getForecast(city)
      ]);

      UI.renderCurrentWeather(current);
      UI.renderForecast(forecast);
      Storage.cacheWeather(city, { current, forecast });
      updateFavButton();
    } catch (error) {
      UI.showError(`Could not find weather for "${city}"`);
    }
  }

  /** Update the favorite button state */
  function updateFavButton() {
    if (favBtn) {
      favBtn.textContent = Storage.isFavorite(currentCity) ? "★" : "☆";
    }
  }

  // Event: Search form submit
  searchForm.addEventListener("submit", (e) => {
    e.preventDefault();
    const city = searchInput.value.trim();
    if (city) loadWeather(city);
  });

  // Event: Toggle favorite
  favBtn?.addEventListener("click", () => {
    if (Storage.isFavorite(currentCity)) {
      Storage.removeFavorite(currentCity);
    } else {
      Storage.addFavorite(currentCity);
    }
    updateFavButton();
    UI.renderFavorites(Storage.getFavorites(), loadWeather);
  });

  // Initial load
  UI.renderFavorites(Storage.getFavorites(), loadWeather);
  loadWeather("London");
});

describe("Storage", () => {
  beforeEach(() => localStorage.clear());

  test("getFavorites returns empty array when no data", () => {
    expect(Storage.getFavorites()).toEqual([]);
  });

  test("addFavorite adds a city", () => {
    Storage.addFavorite("London");
    expect(Storage.getFavorites()).toContain("London");
  });

  test("addFavorite prevents duplicates", () => {
    Storage.addFavorite("London");
    Storage.addFavorite("London");
    expect(Storage.getFavorites().length).toBe(1);
  });

  test("addFavorite limits to 10 cities", () => {
    for (let i = 0; i < 12; i++) Storage.addFavorite(`City${i}`);
    expect(Storage.getFavorites().length).toBe(10);
  });

  test("removeFavorite removes a city", () => {
    Storage.addFavorite("Paris");
    Storage.removeFavorite("Paris");
    expect(Storage.isFavorite("Paris")).toBe(false);
  });

  test("cache returns null for expired data", () => {
    // Manually set old cache
    const old = { data: {}, timestamp: Date.now() - 700000 };
    localStorage.setItem("weather_cache", JSON.stringify({ London: old }));
    expect(Storage.getCachedWeather("London")).toBeNull();
  });
});
