# Blog Database with SQLite
# Schema: users, posts, comments, tags, post_tags (many-to-many)
# Each table has created_at and updated_at timestamps

import sqlite3
from datetime import datetime
from typing import List, Optional, Dict, Any

DATABASE = "blog.db"

def get_connection():
    conn = sqlite3.connect(DATABASE)
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA foreign_keys = ON")
    return conn

def create_tables():
    """Create all tables for the blog database."""
    conn = get_connection()
    cursor = conn.cursor()

    cursor.executescript("""
        CREATE TABLE IF NOT EXISTS users (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            username TEXT UNIQUE NOT NULL,
            email TEXT UNIQUE NOT NULL,
            bio TEXT DEFAULT '',
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        );

        CREATE TABLE IF NOT EXISTS posts (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT NOT NULL,
            content TEXT NOT NULL,
            author_id INTEGER NOT NULL,
            status TEXT DEFAULT 'draft' CHECK(status IN ('draft','published','archived')),
            views INTEGER DEFAULT 0,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (author_id) REFERENCES users(id) ON DELETE CASCADE
        );

        CREATE TABLE IF NOT EXISTS comments (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            post_id INTEGER NOT NULL,
            user_id INTEGER NOT NULL,
            content TEXT NOT NULL,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
            FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
        );

        CREATE TABLE IF NOT EXISTS tags (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT UNIQUE NOT NULL
        );

        CREATE TABLE IF NOT EXISTS post_tags (
            post_id INTEGER NOT NULL,
            tag_id INTEGER NOT NULL,
            PRIMARY KEY (post_id, tag_id),
            FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
            FOREIGN KEY (tag_id) REFERENCES tags(id) ON DELETE CASCADE
        );
    """)

    conn.commit()
    conn.close()
    print("✅ All tables created successfully")

# ==================== CREATE ====================

def create_user(username: str, email: str, bio: str = "") -> int:
    """Create a new user and return their ID."""
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute(
        "INSERT INTO users (username, email, bio) VALUES (?, ?, ?)",
        (username, email, bio)
    )
    conn.commit()
    user_id = cursor.lastrowid
    conn.close()
    return user_id

def create_post(title: str, content: str, author_id: int, 
               status: str = "draft") -> int:
    """Create a new blog post and return its ID."""
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute(
        "INSERT INTO posts (title, content, author_id, status) VALUES (?, ?, ?, ?)",
        (title, content, author_id, status)
    )
    conn.commit()
    post_id = cursor.lastrowid
    conn.close()
    return post_id

def add_comment(post_id: int, user_id: int, content: str) -> int:
    """Add a comment to a post."""
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute(
        "INSERT INTO comments (post_id, user_id, content) VALUES (?, ?, ?)",
        (post_id, user_id, content)
    )
    conn.commit()
    comment_id = cursor.lastrowid
    conn.close()
    return comment_id

def add_tag_to_post(post_id: int, tag_name: str):
    """Add a tag to a post. Creates the tag if it doesn't exist."""
    conn = get_connection()
    cursor = conn.cursor()
    # Insert tag if it doesn't exist
    cursor.execute("INSERT OR IGNORE INTO tags (name) VALUES (?)", (tag_name,))
    # Get the tag ID
    cursor.execute("SELECT id FROM tags WHERE name = ?", (tag_name,))
    tag_id = cursor.fetchone()["id"]
    # Link post and tag
    cursor.execute(
        "INSERT OR IGNORE INTO post_tags (post_id, tag_id) VALUES (?, ?)",
        (post_id, tag_id)
    )
    conn.commit()
    conn.close()
READ Operations
# ==================== READ ====================

def get_post_with_author(post_id: int) -> Optional[Dict]:
    """Get a post with its author info using a JOIN."""
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute("""
        SELECT p.*, u.username, u.email
        FROM posts p
        JOIN users u ON p.author_id = u.id
        WHERE p.id = ?
    """, (post_id,))
    row = cursor.fetchone()
    conn.close()
    return dict(row) if row else None

def get_posts_by_tag(tag_name: str) -> List[Dict]:
    """Get all published posts with a specific tag."""
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute("""
        SELECT p.id, p.title, p.created_at, u.username
        FROM posts p
        JOIN post_tags pt ON p.id = pt.post_id
        JOIN tags t ON pt.tag_id = t.id
        JOIN users u ON p.author_id = u.id
        WHERE t.name = ? AND p.status = 'published'
        ORDER BY p.created_at DESC
    """, (tag_name,))
    rows = cursor.fetchall()
    conn.close()
    return [dict(r) for r in rows]

def get_post_comments(post_id: int) -> List[Dict]:
    """Get all comments for a post with usernames."""
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute("""
        SELECT c.content, c.created_at, u.username
        FROM comments c
        JOIN users u ON c.user_id = u.id
        WHERE c.post_id = ?
        ORDER BY c.created_at ASC
    """, (post_id,))
    rows = cursor.fetchall()
    conn.close()
    return [dict(r) for r in rows]

def search_posts(query: str) -> List[Dict]:
    """Search posts by title or content."""
    conn = get_connection()
    cursor = conn.cursor()
    search = f"%{query}%"
    cursor.execute("""
        SELECT id, title, substr(content, 1, 100) as excerpt, created_at
        FROM posts
        WHERE title LIKE ? OR content LIKE ?
        ORDER BY created_at DESC
    """, (search, search))
    rows = cursor.fetchall()
    conn.close()
    return [dict(r) for r in rows]
UPDATE Operations
# ==================== UPDATE ====================

def update_post(post_id: int, title: str = None, 
               content: str = None, status: str = None) -> bool:
    """Update a post's fields. Only updates provided fields."""
    conn = get_connection()
    updates, params = [], []
    if title: updates.append("title = ?"); params.append(title)
    if content: updates.append("content = ?"); params.append(content)
    if status: updates.append("status = ?"); params.append(status)
    updates.append("updated_at = ?")
    params.append(datetime.now().isoformat())
    params.append(post_id)

    cursor = conn.cursor()
    cursor.execute(
        f"UPDATE posts SET {', '.join(updates)} WHERE id = ?", params
    )
    conn.commit()
    affected = cursor.rowcount
    conn.close()
    return affected > 0

def increment_views(post_id: int):
    """Increment the view count for a post."""
    conn = get_connection()
    conn.execute("UPDATE posts SET views = views + 1 WHERE id = ?", (post_id,))
    conn.commit()
    conn.close()

def get_blog_stats() -> Dict:
    """Get overall blog statistics."""
    conn = get_connection()
    cursor = conn.cursor()

    stats = {}

    # Total counts
    cursor.execute("SELECT COUNT(*) as count FROM users")
    stats["total_users"] = cursor.fetchone()["count"]

    cursor.execute("SELECT COUNT(*) as count FROM posts WHERE status='published'")
    stats["published_posts"] = cursor.fetchone()["count"]

    # Most active author
    cursor.execute("""
        SELECT u.username, COUNT(p.id) as post_count
        FROM users u
        JOIN posts p ON u.id = p.author_id
        GROUP BY u.id
        ORDER BY post_count DESC
        LIMIT 1
    """)
    row = cursor.fetchone()
    stats["top_author"] = dict(row) if row else None

    # Most popular post by views
    cursor.execute("""
        SELECT title, views FROM posts
        ORDER BY views DESC LIMIT 1
    """)
    row = cursor.fetchone()
    stats["most_viewed"] = dict(row) if row else None

    # Popular tags
    cursor.execute("""
        SELECT t.name, COUNT(pt.post_id) as usage_count
        FROM tags t
        JOIN post_tags pt ON t.id = pt.tag_id
        GROUP BY t.id
        ORDER BY usage_count DESC
        LIMIT 5
    """)
    stats["popular_tags"] = [dict(r) for r in cursor.fetchall()]

    conn.close()
    return stats

6 MongoDB Equivalent (Optional)

Now ask Copilot Chat: "Convert this SQLite blog database to MongoDB using pymongo. Keep the same operations but use document-oriented patterns."

blog_mongo.py — Key differences highlighted
from pymongo import MongoClient
from datetime import datetime

client = MongoClient("mongodb://localhost:27017/")
db = client["blog_db"]

# In MongoDB, comments are EMBEDDED inside posts (denormalized)
def create_post_mongo(title, content, author, tags=None):
    """MongoDB stores posts as rich documents with embedded data."""
    post = {
        "title": title,
        "content": content,
        "author": {  # Embedded document instead of foreign key
            "username": author["username"],
            "email": author["email"]
        },
        "tags": tags or [],  # Array field instead of join table
        "comments": [],       # Embedded array instead of separate table
        "views": 0,
        "status": "draft",
        "created_at": datetime.now(),
        "updated_at": datetime.now()
    }
    result = db.posts.insert_one(post)
    return result.inserted_id

# Aggregation pipeline — equivalent to SQL GROUP BY + JOIN
def get_posts_by_tag_mongo(tag_name):
    """MongoDB aggregation replaces SQL JOINs."""
    pipeline = [
        {"$match": {"tags": tag_name, "status": "published"}},
        {"$sort": {"created_at": -1}},
        {"$project": {
            "title": 1,
            "author.username": 1,
            "comment_count": {"$size": "$comments"},
            "created_at": 1
        }}
    ]
    return list(db.posts.aggregate(pipeline))
💡 SQL vs MongoDB Key Differences

SQL: Normalized tables + JOINs + Foreign keys
MongoDB: Denormalized documents + Embedded arrays + Aggregation pipelines
Copilot understands both paradigms and adapts its suggestions based on the database driver you import.
