🗂️ The Messy Code

Create a file called messy_utils.js and paste this deliberately problematic code. It contains bugs, no documentation, poor naming, code smells, and missing error handling.

messy_utils.js — The starting code (paste this) 
// utils - various helper functions
var data = []

function proc(arr) {
  var r = []
  for (var i = 0; i <= arr.length; i++) {  // BUG: off-by-one
    if (arr[i] !== null && arr[i] !== undefined) {
      r.push(arr[i].toString().trim().toLowerCase())
    }
  }
  return r
}

function calc(items) {
  var t = 0
  for(var i=0;i<items.length;i++){
    t += items[i].price * items[i].qty  // BUG: no check for missing fields
  }
  var tax = t * 0.08
  var d = 0
  if(t > 100) d = t * 0.1
  if(t > 500) d = t * 0.15  // BUG: should be else if
  return {total: t, tax: tax, discount: d, final: t + tax - d}
}

function fmt(d) {
  var m = d.getMonth()  // BUG: months are 0-indexed, not handled
  var dy = d.getDate()
  var y = d.getFullYear()
  return m + '/' + dy + '/' + y
}

function srch(arr, k) {
  for(var i=0;i<arr.length;i++){
    if(arr[i].name.toLowerCase() == k.toLowerCase())  // BUG: == not ===, no null check
      return arr[i]
  }
  return null
}

function dedupe(a) {
  var seen = {}
  var out = []
  for(var i=0;i<a.length;i++){
    if(!seen[a[i]]){
      seen[a[i]] = true
      out.push(a[i])
    }
  }
  return out
}

module.exports = {proc, calc, fmt, srch, dedupe}

  2. Type /fix in Copilot Chat

Copilot should find at least these bugs:

  • proc(): Off-by-one error — i <= arr.length should be i < arr.length
  • calc(): Missing null checks for price/qty fields; discount logic should use else if
  • fmt(): Month is 0-indexed — need to add 1; missing zero-padding
  • srch(): Using == instead of ===; no null check on arr[i].name or k
  • All functions: Using var instead of const/let

Accept the fixes and verify the code runs correctly.
