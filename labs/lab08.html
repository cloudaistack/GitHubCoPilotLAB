🗂️ The Undocumented Project

Create a file inventory.py with this undocumented code:

inventory.py — Paste this undocumented code
from datetime import datetime
from typing import Optional
import json

class Product:
    def __init__(self, sku, name, price, quantity=0, category="general"):
        self.sku = sku
        self.name = name
        self.price = price
        self.quantity = quantity
        self.category = category
        self.created_at = datetime.now()
        self.updated_at = datetime.now()

    def update_price(self, new_price):
        if new_price < 0:
            raise ValueError("Price cannot be negative")
        self.price = new_price
        self.updated_at = datetime.now()

    def adjust_stock(self, amount):
        if self.quantity + amount < 0:
            raise ValueError("Insufficient stock")
        self.quantity += amount
        self.updated_at = datetime.now()

    def to_dict(self):
        return {"sku": self.sku, "name": self.name, "price": self.price,
                "quantity": self.quantity, "category": self.category}


class Inventory:
    def __init__(self):
        self.products = {}

    def add_product(self, product):
        if product.sku in self.products:
            raise ValueError(f"SKU {product.sku} already exists")
        self.products[product.sku] = product

    def remove_product(self, sku):
        if sku not in self.products:
            raise KeyError(f"Product {sku} not found")
        del self.products[sku]

    def get_product(self, sku):
        return self.products.get(sku)

    def search(self, query, category=None):
        results = []
        for p in self.products.values():
            if query.lower() in p.name.lower():
                if category is None or p.category == category:
                    results.append(p)
        return results

    def get_low_stock(self, threshold=5):
        return [p for p in self.products.values() if p.quantity <= threshold]

    def total_value(self):
        return sum(p.price * p.quantity for p in self.products.values())

    def export_json(self, filepath):
        data = [p.to_dict() for p in self.products.values()]
        with open(filepath, 'w') as f:
            json.dump(data, f, indent=2)

    def import_json(self, filepath):
        with open(filepath, 'r') as f:
            data = json.load(f)
        for item in data:
            product = Product(**item)
            self.add_product(product)

📝 Step-by-Step Instructions

1Add Docstrings with /doc
  1. Select the Product class (all of it)
  2. In Copilot Chat, type /doc
  3. Review and accept the generated docstrings
  4. Repeat for the Inventory class
  5. Each method should now have: description, parameters with types, return value, raises info, and example
2Generate README.md

In Copilot Chat, type:

Copilot Chat Prompt
@workspace Generate a comprehensive README.md for this project. Include:
- Project title and description
- Features list
- Installation instructions
- Quick start example showing how to create an inventory, add products, and search
- API Reference section documenting all public classes and methods
- Usage examples for each major feature
- Contributing guidelines
- License (MIT)

Save the output as README.md in your project root.

4Add Type Hints and Module Docstring
  1. Select the entire inventory.py file

  3. Review and accept the changes
